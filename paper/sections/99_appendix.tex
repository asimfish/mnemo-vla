\section{Reproducibility Checklist}
\label{app:repro}

\begin{itemize}
\item Code: \url{https://github.com/asimfish/mnemo-vla} (MIT)
\item Tests: \texttt{pytest -q tests/} (20 tests, CPU-only)
\item Benchmark: \texttt{PYTHONPATH=. python examples/retrieval\_benchmark.py}
\item Quickstart: \texttt{PYTHONPATH=. python examples/quickstart.py}
\item Dependencies: Python $\geq$ 3.10, numpy $\geq$ 1.24.
      Optional extras: \texttt{faiss-cpu} for scale,
      \texttt{sentence-transformers} for semantic retrieval.
\item No GPU required for primitives; hot-memory and VLA training
      experiments (deferred to v0.4+) will specify their own
      requirements.
\end{itemize}

\section{Language Memory Schema (Formal Grammar)}
\label{app:grammar}

\begin{verbatim}
memory      := task_line session_line? subtasks? cross_session? failures?
task_line   := "[TASK] " <string>
session_line:= "[SESSION] " <string>
subtasks    := "[SUBTASKS]" subtask_line+
subtask_line:= integer "." marker description details?
marker      := "[X]" | "[>]" | "[ ]" | "[!]"
details     := "(" <string> ")"
cross_session := "[CROSS-SESSION]" fact_line+
fact_line   := "-" content "(source:" session "," "conf=" float ")"
failures    := "[FAILURES]" failure_line+
failure_line:= "-" description ("->" intervention)?
\end{verbatim}
